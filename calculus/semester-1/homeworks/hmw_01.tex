\exercise{5.229}{
На $\epsilon - \delta$ языке
\[
  \boxed{
    ( \lim_{n \to \infty}{x_{n}} = A ) := \forall \epsilon > 0 \   \exists N \in \N \ \forall n > N \ ( | x_{n} - A | < \epsilon )
  }
\]
\\
Через окрестности
\[
  \boxed{
    ( \lim_{n \to \infty}{x_{n}} = A ) := \forall V(A) \ \exists N \in \N \ \forall n > N \ ( x_{n} \in V(A) )
 }
\]
}

\exercise{5.230 (в)}{ Найти $a = \lim_{n\to\infty}{x_n}$ и определить номер $N_\epsilon$ такой, что $|x_n - a| < \epsilon$ при всех $n > N_\epsilon$, если $x_n = \frac{1}{n} \sin{\frac{\pi n}{2}}$, $\epsilon = \frac{1}{1000}$

$$
\lim_{n\to\infty}\frac{1}{n} \sin{\frac{\pi n}{2}} = 0
$$$$
|\frac{1}{n} \sin{\frac{\pi n}{2}}| < \frac{1}{1000}
$$$$
|\frac{1000}{n} \sin{\frac{\pi n}{2}}| < 1
$$
\textbf{Ответ:} $N^{(\epsilon)} > 999; a = 0$
}
